017package com.github.siroshun09.mccommand.paper.listener;
018
019import com.destroystokyo.paper.event.server.AsyncTabCompleteEvent;
020import com.github.siroshun09.mccommand.common.Command;
021import com.github.siroshun09.mccommand.common.argument.Argument;
022import com.github.siroshun09.mccommand.common.context.SimpleCommandContext;
023import com.github.siroshun09.mccommand.paper.sender.PaperSender;
024import org.bukkit.event.EventHandler;
025import org.bukkit.event.HandlerList;
026import org.bukkit.event.Listener;
027import org.bukkit.plugin.Plugin;
028import org.jetbrains.annotations.NotNull;
029
030import java.util.LinkedList;
031import java.util.List;
032import java.util.Objects;
033
034/**
035 * A class that enables asynchronous tab completion.
036 */
037public class AsyncTabCompleteListener implements Listener {
038
039    private final Command command;
040
041    private AsyncTabCompleteListener(@NotNull Command command) {
042        this.command = command;
043    }
044
045    /**
046     * Enables asynchronous tab completion.
047     *
048     * @param plugin  the plugin to register the listener
049     * @param command the {@link Command} for asynchronous tab completion
050     * @return the listener of {@link AsyncTabCompleteEvent}
051     */
052    @NotNull
053    public static AsyncTabCompleteListener register(@NotNull Plugin plugin, @NotNull Command command) {
054        AsyncTabCompleteListener listener = new AsyncTabCompleteListener(command);
055
056        plugin.getServer().getPluginManager().registerEvents(listener, plugin);
057
058        return listener;
059    }
060
061    /**
062     * Unregisters this listener.
063     */
064    public void unregister() {
065        HandlerList.unregisterAll(this);
066    }
067
068    /**
069     * The method called when the {@link AsyncTabCompleteEvent} is fired.
070     *
071     * @param event the {@link AsyncTabCompleteEvent}
072     */
073    @EventHandler
074    public void onTabCompletion(@NotNull AsyncTabCompleteEvent event) {
075        Objects.requireNonNull(event, "event");
076
077        if (!event.isCommand()) {
078            return;
079        }
080
081        String buffer = event.getBuffer();
082        if (buffer.isEmpty()) {
083            return;
084        }
085
086        if (buffer.charAt(0) == '/') {
087            buffer = buffer.substring(1);
088        }
089
090        int firstSpace = buffer.indexOf(' ');
091        if (firstSpace < 0) {
092            return;
093        }
094
095        String alias = buffer.substring(0, firstSpace).toLowerCase();
096
097        if (!command.getName().equals(alias) || !command.getAliases().contains(alias)) {
098            return;
099        }
100
101        String[] args = buffer.substring(firstSpace + 1).split(" ");
102        List<Argument> arguments;
103
104        if (args.length == 0) {
105            arguments = List.of(Argument.of(0, ""));
106        } else if (!args[0].equals("") && buffer.endsWith(" ")) {
107            List<Argument> temp = createArguments(new LinkedList<>(), args);
108
109            temp.add(Argument.of(args.length, ""));
110
111            arguments = List.copyOf(temp);
112        } else {
113            arguments = List.copyOf(createArguments(new LinkedList<>(), args));
114        }
115
116        event.setCompletions(command.onTabCompletion(
117                SimpleCommandContext.newBuilder()
118                        .setCommand(command)
119                        .setSender(new PaperSender(event.getSender()))
120                        .setArguments(arguments)
121                        .setLabel(alias)
122                        .build()
123        ));
124
125        event.setHandled(true);
126    }
127
128    @NotNull
129    private List<Argument> createArguments(@NotNull List<Argument> toCollect, @NotNull String[] args) {
130        for (int i = 0; i < args.length; i++) {
131            toCollect.add(Argument.of(i, args[i]));
132        }
133
134        return toCollect;
135    }
136}
