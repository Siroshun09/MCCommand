017package com.github.siroshun09.mccommand.common.filter;
018
019import org.jetbrains.annotations.Contract;
020import org.jetbrains.annotations.NotNull;
021
022import java.util.function.Predicate;
023
024/**
025 * A class that implements {@link Filter} for the {@link Number}.
026 *
027 * @param <N> the {@link Number} type
028 */
029public final class NumberFilter<N extends Number> extends AbstractFilter<N> {
030
031    /**
032     * The constructor of {@link NumberFilter}
033     *
034     * @param predicate the {@link Predicate} to test the value
035     */
036    private NumberFilter(@NotNull Predicate<N> predicate) {
037        super(predicate);
038    }
039
040    /**
041     * Creates the {@link NumberFilter}.
042     *
043     * @param predicate the predicate to test the {@link Number}
044     * @param <N>       the {@link Number} type
045     * @return the {@link NumberFilter}
046     */
047    @Contract("_ -> new")
048    public static <N extends Number> @NotNull NumberFilter<N> create(@NotNull Predicate<N> predicate) {
049        return new NumberFilter<N>(predicate);
050    }
051}
